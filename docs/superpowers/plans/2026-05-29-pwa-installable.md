# Recycle ERP — Installable PWA Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the existing React/Vite SPA into an installable PWA with mobile-grade polish — Add-to-Home-Screen on iOS/Android, an in-app install prompt, an update-available flow, resilient background uploads, and a Web Share Target that routes shared photos into the AI label flow.

**Architecture:** Add `vite-plugin-pwa` (Workbox-based) to the existing Vite build. Use Workbox `generateSW` (no custom SW source) — the precache manifest is auto-generated from the build, runtime caching rules are declared in `vite.config.ts`. SW scope is the whole origin **except** `/v/*` (vendor portal — short-lived token URLs, not the operator app). Auth and data calls stay network-first; offline reads/writes are explicitly out of scope (deferred to a later "Tier 3" plan). Caddy in front of the SPA sets `Cache-Control: no-cache` for the SW/manifest so updates land within one reload.

**Tech Stack:** Vite 6 · React 18 · `vite-plugin-pwa@^0.21` (Workbox 7) · Caddy (prod) · existing frontend toolchain (TypeScript, pnpm). No backend changes. No new runtime deps beyond `vite-plugin-pwa` (dev) and `workbox-window` (runtime, pulled in transitively).

```ts
// Service-worker registration + update bridge.
//
// We use vite-plugin-pwa's virtual `registerType: 'prompt'` flow: the SW
// installs in the background and waits in `installing` until the user
// accepts an update. The PwaUpdateToast component (Task 9) listens for the
// 'pwa:needRefresh' event and calls `applyUpdate()` when the user clicks.

import { registerSW } from 'virtual:pwa-register';

let applyUpdateFn: (() => Promise<void>) | null = null;

export function registerPwa(): void {
  if (typeof window === 'undefined') return;
  if (!('serviceWorker' in navigator)) return;
  // The vendor portal is its own short-lived URL space; skip SW there.
  if (window.location.pathname.startsWith('/v/')) return;

  const updateSW = registerSW({
    onNeedRefresh() {
      applyUpdateFn = async () => {
        await updateSW(true); // reload after the new SW activates
      };
      window.dispatchEvent(new CustomEvent('pwa:needRefresh'));
    },
    onOfflineReady() {
      window.dispatchEvent(new CustomEvent('pwa:offlineReady'));
    },
  });
}

export function applyPwaUpdate(): Promise<void> | void {
  if (applyUpdateFn) return applyUpdateFn();
}
```

## Task 6: Caddyfile headers for SW and manifest

**Files:**
- Modify: `apps/frontend/Caddyfile`

The default Caddy `file_server` serves all static files with the same caching policy. PWAs are sensitive: `sw.js` must **not** be long-cached, otherwise users get stuck on an old SW for days. Workbox precache files (`workbox-<hash>.js`) **should** be long-cached (they're content-hashed). The manifest should be revalidated.

- [ ] **Step 6.1: Add header rules**

Edit `apps/frontend/Caddyfile`. Inside the existing `:80 { ... }` block, before the existing `handle { ... }` SPA block, add:

```caddy
	# Service Worker: must not be cached by browsers — updates land within one reload.
	@sw path /sw.js
	header @sw {
		Cache-Control "no-cache"
		Service-Worker-Allowed "/"
	}

	# Web manifest: revalidate so manifest edits propagate quickly.
	@manifest path /manifest.webmanifest
	header @manifest Cache-Control "no-cache"

	# Workbox runtime + precache JS: content-hashed, safe to cache hard.
	@workbox path_regexp ^/workbox-[a-f0-9]+\.js$
	header @workbox Cache-Control "public, max-age=31536000, immutable"
```

```tsx
import { useEffect, useState } from 'react';
import { useT } from '../lib/i18n';

// Chrome's BeforeInstallPromptEvent isn't in lib.dom.d.ts.
interface BeforeInstallPromptEvent extends Event {
  prompt(): Promise<void>;
  userChoice: Promise<{ outcome: 'accepted' | 'dismissed'; platform: string }>;
}

const DISMISS_KEY = 'pwa:installDismissedAt';
const DISMISS_TTL_MS = 1000 * 60 * 60 * 24 * 14; // 14 days

function recentlyDismissed(): boolean {
  try {
    const raw = localStorage.getItem(DISMISS_KEY);
    if (!raw) return false;
    return Date.now() - Number(raw) < DISMISS_TTL_MS;
  } catch {
    return false;
  }
}

function isStandalone(): boolean {
  return window.matchMedia?.('(display-mode: standalone)').matches
    || (window.navigator as { standalone?: boolean }).standalone === true;
}

function isIos(): boolean {
  return /iPad|iPhone|iPod/.test(window.navigator.userAgent)
    && !(window.navigator as { MSStream?: unknown }).MSStream;
}

export function PwaInstallPrompt() {
  const { t } = useT();
  const [evt, setEvt] = useState<BeforeInstallPromptEvent | null>(null);
  const [showIosHint, setShowIosHint] = useState(false);

  useEffect(() => {
    if (isStandalone() || recentlyDismissed()) return;

    const onBeforeInstall = (e: Event) => {
      e.preventDefault();
      setEvt(e as BeforeInstallPromptEvent);
    };
    window.addEventListener('beforeinstallprompt', onBeforeInstall);

    if (isIos()) setShowIosHint(true);

    return () => window.removeEventListener('beforeinstallprompt', onBeforeInstall);
  }, []);

  const accept = async () => {
    if (!evt) return;
    await evt.prompt();
    await evt.userChoice;
    setEvt(null);
  };

  const dismiss = () => {
    try { localStorage.setItem(DISMISS_KEY, String(Date.now())); } catch { /* ignore */ }
    setEvt(null);
    setShowIosHint(false);
  };

  if (evt) {
    return (
      <div className="pwa-install-prompt" role="dialog" aria-live="polite">
        <button type="button" className="pwa-install-cta" onClick={accept}>
          {t('pwa.install.cta')}
        </button>
        <button type="button" className="pwa-install-dismiss" onClick={dismiss}>
          {t('pwa.install.dismiss')}
        </button>
      </div>
    );
  }
  if (showIosHint) {
    return (
      <div className="pwa-install-prompt pwa-install-prompt--ios" role="status">
        <span>{t('pwa.install.iosHint')}</span>
        <button type="button" className="pwa-install-dismiss" onClick={dismiss}>
          {t('pwa.install.dismiss')}
        </button>
      </div>
    );
  }
  return null;
}
```

## Task 10: Background-sync for attachment uploads

**Files:**
- Modify: `apps/frontend/vite.config.ts`
- Modify: `apps/frontend/src/lib/i18n.tsx` (optional offline toast)

The mobile flow uploads photos to `/api/attachments` (label scans, sell-order attachments). Today, a failed POST on a flaky cellular link just errors out — the warehouse user has to retry manually. Workbox `BackgroundSyncPlugin` lets us queue the failed request and replay it when the browser fires the `sync` event.

Important: this **only** applies to `POST /api/attachments`. We don't want to retry every API call; mutations to status-guarded routes (PO/SO transitions) would be dangerous to replay blindly because they'd race with what the user did online after coming back.

- [ ] **Step 10.1: Add the background-sync rule**

In `apps/frontend/vite.config.ts`, inside the `runtimeCaching` array, **before** the catch-all `/api/` `NetworkOnly` rule, add:

```ts
{
  // Background-sync only for attachment uploads; the queue retries when
  // connectivity returns. Other mutations (status changes, etc.) must NOT
  // be auto-replayed — they could race with what the user did since.
  urlPattern: ({ url, request }) =>
    url.pathname === '/api/attachments' && request.method === 'POST',
  method: 'POST',
  handler: 'NetworkOnly',
  options: {
    backgroundSync: {
      name: 'recycle-erp-attachments',
      options: {
        maxRetentionTime: 60 * 24, // minutes — drop after 24h
      },
    },
  },
},
```

(Workbox requires `method: 'POST'` for POST handlers — by default it only matches GET.)

```tsx
import { useEffect } from 'react';
import { navigate } from '../lib/route';

// Web Share Target endpoint. The browser POSTs the shared file(s) here as
// multipart/form-data; we stash the first image in sessionStorage as a
// blob URL and forward to the submit flow, which reads it on mount and
// drops it into the AI label dropzone.
//
// This component renders nothing visible — it's a redirect with side
// effects. The actual POST is intercepted by the service worker in dev
// (vite-plugin-pwa) and in prod (Workbox precache fallback).

const SHARED_FILE_KEY = 'pwa:sharedFile';

export function ShareTarget() {
  useEffect(() => {
    (async () => {
      // The SW exposes the multipart body via navigator.serviceWorker.controller
      // OR via window.location if the browser used a GET fallback. For multipart
      // POST, vite-plugin-pwa relays it through a message event.
      //
      // Simpler path: form submission lands here as a navigation; the file is
      // attached to the SW's request. We use the Service Worker's `clients`
      // API by listening for a message from the SW containing the file.
      const file = await new Promise<File | null>((resolve) => {
        const handler = (e: MessageEvent) => {
          if (e.data?.type === 'pwa:sharedFile' && e.data.file instanceof File) {
            navigator.serviceWorker.removeEventListener('message', handler);
            resolve(e.data.file);
          }
        };
        navigator.serviceWorker?.addEventListener('message', handler);
        // Tell the SW we're ready.
        navigator.serviceWorker?.controller?.postMessage({ type: 'pwa:claimSharedFile' });
        // Time-out after 2s — if nothing arrives, this was a stale navigation.
        setTimeout(() => {
          navigator.serviceWorker?.removeEventListener('message', handler);
          resolve(null);
        }, 2000);
      });

      if (file) {
        try {
          sessionStorage.setItem(SHARED_FILE_KEY, await blobToDataUrl(file));
        } catch { /* ignore quota */ }
      }
      navigate('/submit');
    })();
  }, []);

  return null;
}

function blobToDataUrl(blob: Blob): Promise<string> {
  return new Promise((resolve, reject) => {
    const r = new FileReader();
    r.onload = () => resolve(String(r.result));
    r.onerror = () => reject(r.error);
    r.readAsDataURL(blob);
  });
}
```

```ts
/// <reference lib="webworker" />
import { precacheAndRoute, createHandlerBoundToURL } from 'workbox-precaching';
import { registerRoute, NavigationRoute, Route } from 'workbox-routing';
import { NetworkOnly, StaleWhileRevalidate, CacheFirst } from 'workbox-strategies';
import { BackgroundSyncPlugin } from 'workbox-background-sync';
import { ExpirationPlugin } from 'workbox-expiration';

declare const self: ServiceWorkerGlobalScope & {
  __WB_MANIFEST: Array<{ url: string; revision: string | null }>;
};

precacheAndRoute(self.__WB_MANIFEST);

// SPA fallback, except for vendor / api / oauth / well-known.
const navRoute = new NavigationRoute(createHandlerBoundToURL('/index.html'), {
  denylist: [/^\/v\//, /^\/api\//, /^\/oauth\//, /^\/\.well-known\//, /^\/share-target$/],
});
registerRoute(navRoute);

// API NetworkOnly + attachment background sync.
const attachmentQueue = new BackgroundSyncPlugin('recycle-erp-attachments', {
  maxRetentionTime: 24 * 60,
});
registerRoute(
  ({ url, request }) =>
    url.pathname === '/api/attachments' && request.method === 'POST',
  new NetworkOnly({ plugins: [attachmentQueue] }),
  'POST',
);
registerRoute(({ url }) => url.pathname.startsWith('/api/'), new NetworkOnly());
registerRoute(({ url }) => url.pathname.startsWith('/oauth/'), new NetworkOnly());
registerRoute(({ url }) => url.pathname.startsWith('/.well-known/'), new NetworkOnly());

// Google Fonts.
registerRoute(
  ({ url }) => url.origin === 'https://fonts.googleapis.com',
  new StaleWhileRevalidate({ cacheName: 'google-fonts-stylesheets' }),
);
registerRoute(
  ({ url }) => url.origin === 'https://fonts.gstatic.com',
  new CacheFirst({
    cacheName: 'google-fonts-webfonts',
    plugins: [new ExpirationPlugin({ maxEntries: 16, maxAgeSeconds: 60 * 60 * 24 * 365 })],
  }),
);

// Web Share Target: intercept POST /share-target, stash the file, hand to
// the page via postMessage once it asks for it.
let pendingSharedFile: File | null = null;

self.addEventListener('fetch', (event) => {
  const url = new URL(event.request.url);
  if (event.request.method === 'POST' && url.pathname === '/share-target') {
    event.respondWith((async () => {
      try {
        const form = await event.request.formData();
        const f = form.get('files');
        if (f instanceof File) pendingSharedFile = f;
      } catch { /* ignore */ }
      // Redirect into the SPA at /share-target so the React page mounts.
      return Response.redirect('/share-target?via=sw', 303);
    })());
  }
});

self.addEventListener('message', (event) => {
  if (event.data?.type === 'pwa:claimSharedFile' && pendingSharedFile) {
    const file = pendingSharedFile;
    pendingSharedFile = null;
    event.source?.postMessage({ type: 'pwa:sharedFile', file });
  }
});

self.addEventListener('install', () => self.skipWaiting());
self.addEventListener('activate', (event) => event.waitUntil(self.clients.claim()));
```
